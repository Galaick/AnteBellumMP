#Events relating to the political instability of Kilwa in the 15th century.
#Coup of Suleiman and Muhammad al-Mazlum.
country_event = {
	id = flavor_zan.5
	title = flavor_zan.5.t
	desc = flavor_zan.5.d
	picture = DEATH_OF_HEIR_eventPicture

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		tag = ZAN #Will trigger upon first ruler change.
		OR = {
			government = monarchy
			government = tribal
		}
		has_government_attribute = heir
	}
	
	option = {
		name = flavor_zan.5.a #Suleiman takes the throne.
		ai_chance = { factor = 1 }
		set_country_flag = zan_plotting_viziers
		add_stability = -2
		define_ruler = {
			name = "Suleiman"
			dynasty = al-Mawahib
			claim = 80
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		#add_legitimacy = -20
		set_ruler_flag = zan_muhammad_and_suleiman
		hidden_effect = {
			define_heir = {
				claim = 75
				dynasty = al-Mazlum
				name = "Muhammad"
				age = 45
			}
		}
	}
	option = {
		name = flavor_zan.5.b #Muhammad al-Muzlum takes the throne.
		ai_chance = { factor = 0 }
		set_country_flag = zan_plotting_viziers
		add_stability = -2
		define_ruler = {
			name = "Muhammad"
			dynasty = "al-Mazlum"
			claim = 80
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		#add_legitimacy = -20
		set_ruler_flag = zan_muhammad_and_suleiman
		hidden_effect = {
			define_heir = {
				claim = 75
				dynasty = al-Mawahib
				name = "Suleiman"
				age = 50
			}
		}
	}
	option = {
		name = flavor_zan.5.c #XXXX is our rightful ruler!
		ai_chance = { factor = 0 }
		set_country_flag = zan_plotting_viziers
		add_stability = -1
		capital_scope = {
			spawn_rebels = {
				type = pretender_rebels
				leader = "Muhammad"
				leader_dynasty = "al-Mazlum"
				size = 2
			}
		}
	}
}

#Mogadishu admits defeat
country_event = {
	id = flavor_zan.20
	title = flavor_zan.20.t
	desc = flavor_zan.20.desc
	picture = CONQUEST_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavor_zan.20.a
		1196 = {
			if = {
				limit = { NOT = { province_has_center_of_trade_of_level = 2 } }
				add_center_of_trade_level = 1
			}
			else = {
				add_base_production = 3
			}
			add_prosperity = 100
		}
		1205 = {
			if = {
				limit = { province_has_center_of_trade_of_level = 2 }
				add_center_of_trade_level = -1
			}
			else = {
				add_base_production = -2
			}
		}
		add_treasury = 400

		ai_chance = {
			factor = 1
		}
	}
}
